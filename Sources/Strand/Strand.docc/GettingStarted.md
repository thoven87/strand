# Getting started

Add Strand to a Swift server project in three steps.

## 1. Add the package

In `Package.swift`:

```swift
dependencies: [
    .package(url: "https://github.com/thoven87/strand", from: "0.1.0"),
],
targets: [
    .target(name: "MyApp", dependencies: [
        .product(name: "Strand", package: "strand"),
    ]),
]
```

## 2. Apply the schema

```bash
psql "postgresql://user:pass@localhost:5432/mydb" -f strand.sql
```

The schema creates the `strand` Postgres schema with all required tables.

## 3. Define activities, a workflow, and a worker

```swift
import Strand
import ServiceLifecycle

// ── I/O types ────────────────────────────────────────────────────────────────

struct EmailInput:  Codable, Sendable { let to: String; let subject: String }
struct EmailResult: Codable, Sendable { let messageID: String }

// ── Activities ────────────────────────────────────────────────────────────────
//
// @ActivityContainer groups related activities that share a common dependency
// (here: the SMTP client). @Activity on each method generates a nested
// Activity-conforming type — no boilerplate struct needed.
//
// Generated type: EmailActivities.SendEmail
// Register with:  activityContainers: [EmailActivities(smtp: smtpClient)]

@ActivityContainer
struct EmailActivities {
    let smtp: SMTPClient   // shared dependency injected once

    @Activity
    func sendEmail(
        input: EmailInput,
        context: ActivityContext
    ) async throws -> EmailResult {
        let id = try await smtp.send(to: input.to, subject: input.subject)
        return EmailResult(messageID: id)
    }
}

// ── Workflow: orchestrates activities ─────────────────────────────────────────
//
// @Workflow synthesises Workflow conformance, handleSignal dispatch for any
// @WorkflowSignal methods, and init() when stored properties have defaults.
// The struct does NOT explicitly list Workflow in its inheritance clause —
// the macro adds it via an extension.

@Workflow
struct WelcomeWorkflow {
    typealias Input  = String   // email address
    typealias Output = String   // message ID

    mutating func run(
        context: WorkflowContext<Self>,
        input: String
    ) async throws -> String {
        // EmailActivities.SendEmail is the Activity type generated by @Activity.
        let result = try await context.runActivity(
            EmailActivities.SendEmail.self,
            input: EmailInput(to: input, subject: "Welcome!")
        )
        return result.messageID
    }
}

// ── Bootstrap with StrandService ─────────────────────────────────────────────

let postgres = PostgresClient(configuration: .init(
    host: "localhost", port: 5432,
    username: "user", password: "pass", database: "mydb",
    tls: .disable
))

var strand = StrandService(
    postgres: postgres,
    options: .init(
        queues: [
            StrandService.QueueConfig(
                name: "default",
                workflows: [WelcomeWorkflow.self],
                // activityContainers lets @ActivityContainer types inject
                // shared dependencies (e.g. SMTP client) at registration time.
                activityContainers: [EmailActivities(smtp: smtpClient)]
            )
        ]
    )
)

// StrandService wires the worker internally — get a client for enqueueing:
let client = strand.client(queue: "default")

let group = ServiceGroup(
    configuration: .init(
        services: [.init(service: postgres), .init(service: strand)],
        gracefulShutdownSignals: [.sigterm, .sigint]
    )
)

// Enqueue a workflow — returns a handle you can poll for the result:
let handle = try await client.startWorkflow(
    WelcomeWorkflow.self,
    input: "alice@example.com"
)

// Optionally await the result:
let messageID = try await handle.result(timeout: .seconds(30))

try await group.run()
```
